import java.util.Stack;

class Solution {
boolean solution(String s) {
boolean answer = true;

        Stack<String> stack = new Stack();

        for (String c : s.split("")) {
            if ("(".equals(c)) {
                stack.add(c);
            } else if (")".equals(c)) {
                if(stack.isEmpty()) return false;
                stack.pop();
            }
        }

        if(!stack.isEmpty()) return false;
        return answer;
    }
}
