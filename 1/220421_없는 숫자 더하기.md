```Java
class Solution {
    public int solution(int[] numbers) {
        int answer = 0;
        boolean[] a = new boolean[10];
        for(int i : numbers){
            a[i] = true;
        }
      
        for(int j=0;j<a.length;j++){
            if(a[j] == false){
                answer += j;
            }
        }
        
        return answer;
    }
}
```
